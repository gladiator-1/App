    /**
     * Takes in a URL and checks if the file extension is a video
     * that can be rendered inside an HTML video element 
     * @param {String} url
     * @param {String} [fileName]
     * @returns {String}
     */
    isVideo(url, fileName) {
        const videoName =  fileName ? fileName : url.includes('data-expensify-source=')? url.split('</a>')[0].split('">')[1]:'';
        const videoExtension = this.getExtension(videoName);
        if(_.contains(['mp4', 'mov'], videoExtension)){
            return "mp4";
        }
        if(_.contains(['webm'], videoExtension)){
            return "webm";
        }
        if(_.contains(['ogg'], videoExtension)){
            return "ogg";
        }
        return "";
    },
